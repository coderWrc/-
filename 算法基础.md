## [多项式输出](https://www.luogu.com.cn/problem/P1067)  
```
#include <iostream>

using namespace std;

int main()
{
	int n; cin >> n;
	for (int i = n; i >= 0; i--)
	{
		int a; cin >> a;

		if (a == 0) continue;

		if (a < 0) cout << '-';
		else if (i != n) cout << '+';

		if (abs(a) != 1 || i == 0) cout << abs(a);

		if (i == 0) continue;
		else if (i == 1) cout << 'x';
		else cout << "x^" << i;
		
	}

	return 0;
}
```
